
import React, { useState } from "react";
import { View, Text, TextInput, Button, ScrollView, StyleSheet } from "react-native";

export default function App() {
  const BASE_URL = "https://your-bot-name.username.repl.co"; // 🔹 استبدل هذا بالرابط من Replit
  const [result, setResult] = useState("");

  const testServer = async () => {
    const res = await fetch(`${BASE_URL}/`);
    const data = await res.json();
    setResult(JSON.stringify(data, null, 2));
  };

  const promoteJob = async () => {
    const res = await fetch(`${BASE_URL}/jobs/promote`, {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ query: "منتج جديد", channel: "Facebook", channel_id: "12345" }),
    });
    const data = await res.json();
    setResult(JSON.stringify(data, null, 2));
  };

  const generatePlan = async () => {
    const res = await fetch(`${BASE_URL}/plans/generate`, {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ niche: "ملابس", budget: 200, days: 10 }),
    });
    const data = await res.json();
    setResult(JSON.stringify(data, null, 2));
  };

  const createPayment = async () => {
    const res = await fetch(`${BASE_URL}/payments/paypal/create`, {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ amount: 50, currency: "USD" }),
    });
    const data = await res.json();
    setResult(JSON.stringify(data, null, 2));
  };

  return (
    <ScrollView contentContainerStyle={styles.container}>
      <Text style={styles.title}>🤖 روبوت التسويق</Text>

      <Button title="اختبار السيرفر" onPress={testServer} />
      <Button title="نشر منتج" onPress={promoteJob} />
      <Button title="خطة تسويقية" onPress={generatePlan} />
      <Button title="إنشاء دفع PayPal" onPress={createPayment} />

      <Text style={styles.result}>{result}</Text>
    </ScrollView>
  );
}

const styles = StyleSheet.create({
  container: {
    flexGrow: 1,
    justifyContent: "center",
    alignItems: "center",
    padding: 20,
    backgroundColor: "#f8f9fa",
  },
  title: {
    fontSize: 22,
    fontWeight: "bold",
    marginBottom: 20,
  },
  result: {
    marginTop: 20,
    padding: 10,
    backgroundColor: "#e9ecef",
    borderRadius: 10,
    width: "100%",
  },
});
const BASE_URL = "https://your-bot-name.username.repl.co";
